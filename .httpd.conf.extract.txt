# Apache httpd.conf extract providing URL routing and other directives

#   Example usage in httpd.conf:
#   
#   # Committee listings system
#   Include /path/to/committee-listings/.httpd.conf.extract.txt
#   Use MacroCommitteeListings "/committees" "/path/to/committee-listings/:/path/to/php/libraries/"


<Macro MacroCommitteeListings $applicationBaseUrl $includePath>
	
	<Location $applicationBaseUrl>
		php_value include_path "$includePath"
	</Location>
	
	Use MacroFrontControllerApplication "$applicationBaseUrl/" "Committee listings"
	
	# Editing
	Use MacroSinenomineEmbeddedWholeDb "$applicationBaseUrl" "/data" "editing"
	
	# Application directives
	RewriteRule ^$applicationBaseUrl/(membership)/$ $applicationBaseUrl/index.html?action=$1 [L]
	RedirectMatch permanent ^$applicationBaseUrl/(membership)$ $applicationBaseUrl/$1/
	RewriteRule ^$applicationBaseUrl/(import)/$ $applicationBaseUrl/index.html?action=$1 [L]
	RedirectMatch permanent ^$applicationBaseUrl/(import)$ $applicationBaseUrl/$1/
	RewriteRule ^$applicationBaseUrl/([a-z]+)/$ $applicationBaseUrl/index.html?action=show&item=$1 [L]
	RedirectMatch permanent ^$applicationBaseUrl/([a-z]+)$ $applicationBaseUrl/$1/
	RewriteRule ^$applicationBaseUrl/([a-z]+)/([0-9]{6})/$ $applicationBaseUrl/index.html?action=meeting&item=$1&date6=$2&page=edit [L]
	RedirectMatch permanent ^$applicationBaseUrl/([a-z]+)$ $applicationBaseUrl/$1/
	RewriteRule ^$applicationBaseUrl/([a-z]+)/([0-9]{6})/(add|delete).html$ $applicationBaseUrl/index.html?action=meeting&item=$1&date6=$2&page=$3 [L]
	
</Macro>
